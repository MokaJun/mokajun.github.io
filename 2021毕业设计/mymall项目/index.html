  3. 用MP的generator逆向工程生成对应类

package com.mokajun.mymall;

import com.baomidou.mybatisplus.annotation.DbType;
import com.baomidou.mybatisplus.annotation.FieldFill;
import com.baomidou.mybatisplus.generator.AutoGenerator;
import com.baomidou.mybatisplus.generator.config.DataSourceConfig;
import com.baomidou.mybatisplus.generator.config.GlobalConfig;
import com.baomidou.mybatisplus.generator.config.PackageConfig;
import com.baomidou.mybatisplus.generator.config.StrategyConfig;
import com.baomidou.mybatisplus.generator.config.po.TableFill;
import com.baomidou.mybatisplus.generator.config.rules.NamingStrategy;

import java.util.ArrayList;
import java.util.List;

public class GeneratorEntry {
    public static void main(String[] args) {
        //创建generator对象
        AutoGenerator autoGenerator = new AutoGenerator();
        //数据源
        DataSourceConfig dataSourceConfig = new DataSourceConfig();
        dataSourceConfig.setDbType(DbType.MYSQL);
        dataSourceConfig.setUrl("jdbc:mysql://localhost:3306/mymall?useUnicode=true&characterEncoding=UTF-8&serverTimeZone=GMT%2B8");
        dataSourceConfig.setUsername("root");
        dataSourceConfig.setPassword("root");
        dataSourceConfig.setDriverName("com.mysql.cj.jdbc.Driver");
        autoGenerator.setDataSource(dataSourceConfig);
        //全局配置
        GlobalConfig globalConfig = new GlobalConfig();
        globalConfig.setOutputDir(System.getProperty("user.dir")+"/src/main/java");
        globalConfig.setOpen(false);//是否打开生成文件夹
        globalConfig.setAuthor("mokajun");//作者
        globalConfig.setServiceName("%sService");//默认名字是IService，这个操作去掉开头的I
        autoGenerator.setGlobalConfig(globalConfig);
        //包信息
        PackageConfig packageConfig = new PackageConfig();
        packageConfig.setParent("com.mokajun.mymall");//父包
//        packageConfig.setModuleName("generator");//Module包，下面的包都在父包/Module包/里,可不写
        packageConfig.setController("controller");
        packageConfig.setService("service");
        packageConfig.setServiceImpl("service.impl");
        packageConfig.setMapper("mapper");
        packageConfig.setEntity("entity");
        autoGenerator.setPackageInfo(packageConfig);
        //配置策略
        StrategyConfig strategyConfig = new StrategyConfig();
        //strategyConfig.setInclude("表名",...);//只添加库中单个或几个表
        strategyConfig.setEntityLombokModel(true);//自动添加lombok注解
        strategyConfig.setNaming(NamingStrategy.underline_to_camel);//下划线转驼峰
        strategyConfig.setColumnNaming(NamingStrategy.underline_to_camel);//下划线转驼峰
        /*
            要自动填充的字段加上注解
            也可以生成后自己手动加，up to you
         */
        List<TableFill> list = new ArrayList<>();
        TableFill tableFill1 = new TableFill("create_time", FieldFill.INSERT);
        TableFill tableFill2 = new TableFill("update_time", FieldFill.INSERT_UPDATE);
        list.add(tableFill1);
        list.add(tableFill2);
        strategyConfig.setTableFillList(list);

        autoGenerator.setStrategy(strategyConfig);
        autoGenerator.execute();
    }
}

  1. ProductCategoryServiceImpl

    实现ProductCategoryVO类从Repository的映射

    package com.mokajun.mymall.service.impl;

import com.baomidou.mybatisplus.core.conditions.query.QueryWrapper;
import com.mokajun.mymall.entity.ProductCategory;
import com.mokajun.mymall.mapper.ProductCategoryMapper;
import com.mokajun.mymall.service.ProductCategoryService;
import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.mokajun.mymall.vo.ProductCategoryVO;
import org.springframework.beans.BeanUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;

/**
 * <p>
 *  服务实现类
 * </p>
 *
 * @author mokajun
 * @since 2021-06-12
 */
@Service
public class ProductCategoryServiceImpl extends ServiceImpl<ProductCategoryMapper, ProductCategory> implements ProductCategoryService {
    @Autowired
    private ProductCategoryMapper productCategoryMapper;

    @Override
    public List<ProductCategoryVO> getAllProductCategoryVO() {
        /*
            实体类转VO
        * */
        //获得实体类
        List<ProductCategoryVO> L1 = new ArrayList<>();
        ProductCategoryVO VO1 = null;
        ProductCategoryVO VO2 = null;
        ProductCategoryVO VO3 = null;
        QueryWrapper wrapper = new QueryWrapper();
        wrapper.eq("type",1);
        List<ProductCategory> productCategoryList = productCategoryMapper.selectList(wrapper);
        for (ProductCategory type1 : productCategoryList) {
            VO1 = new ProductCategoryVO();
            BeanUtils.copyProperties(type1,VO1);
            //添加children
            List<ProductCategoryVO> childrenVOList1 = new ArrayList<>();
            QueryWrapper wrapper1 = new QueryWrapper();
            wrapper1.eq("parent_id",VO1.getId());
            List<ProductCategory> childrenList1 = productCategoryMapper.selectList(wrapper1);
            for (ProductCategory type2 : childrenList1) {
                VO2 = new ProductCategoryVO();
                BeanUtils.copyProperties(type2,VO2);
                //添加children
                List<ProductCategoryVO> childrenVOList2 = new ArrayList<>();
                QueryWrapper wrapper2 = new QueryWrapper();
                wrapper2.eq("parent_id",VO2.getId());
                List<ProductCategory> childrenList2 = productCategoryMapper.selectList(wrapper2);
                for (ProductCategory type3 : childrenList2) {
                    VO3 = new ProductCategoryVO();
                    BeanUtils.copyProperties(type3,VO3);
                    //添加children
                    VO3.setChildren(null);
                    childrenVOList2.add(VO3);
                }

                VO2.setChildren(childrenVOList2);
                childrenVOList1.add(VO2);
            }
            VO1.setChildren(childrenVOList1);
            L1.add(VO1);
        }

        return L1;
    }
}
