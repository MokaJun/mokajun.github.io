@Service
public class ProductCategoryServiceImpl extends ServiceImpl<ProductCategoryMapper, ProductCategory> implements ProductCategoryService {
    @Autowired
    private ProductCategoryMapper productCategoryMapper;

    @Override
    public List<ProductCategoryVO> getAllProductCategoryVO() {
        /*
            å®ä½“ç±»è½¬VO
        * */
        //è·å¾—å®ä½“ç±»
        List<ProductCategoryVO> L1 = new ArrayList<>();
        ProductCategoryVO VO1 = null;
        ProductCategoryVO VO2 = null;
        ProductCategoryVO VO3 = null;
        QueryWrapper wrapper = new QueryWrapper();
        wrapper.eq("type",1);
        List<ProductCategory> productCategoryList = productCategoryMapper.selectList(wrapper);
        for (ProductCategory type1 : productCategoryList) {
            VO1 = new ProductCategoryVO();
            BeanUtils.copyProperties(type1,VO1);
            //æ·»åŠ children
            List<ProductCategoryVO> childrenVOList1 = new ArrayList<>();
            QueryWrapper wrapper1 = new QueryWrapper();
            wrapper1.eq("parent_id",VO1.getId());
            List<ProductCategory> childrenList1 = productCategoryMapper.selectList(wrapper1);
            for (ProductCategory type2 : childrenList1) {
                VO2 = new ProductCategoryVO();
                BeanUtils.copyProperties(type2,VO2);
                //æ·»åŠ children
                List<ProductCategoryVO> childrenVOList2 = new ArrayList<>();
                QueryWrapper wrapper2 = new QueryWrapper();
                wrapper2.eq("parent_id",VO2.getId());
                List<ProductCategory> childrenList2 = productCategoryMapper.selectList(wrapper2);
                for (ProductCategory type3 : childrenList2) {
                    VO3 = new ProductCategoryVO();
                    BeanUtils.copyProperties(type3,VO3);
                    //æ·»åŠ children
                    VO3.setChildren(null);
                    childrenVOList2.add(VO3);
                }

                VO2.setChildren(childrenVOList2);
                childrenVOList1.add(VO2);
            }
            VO1.setChildren(childrenVOList1);
            L1.add(VO1);
        }

        return L1;
    }
}
