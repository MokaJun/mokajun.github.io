快速上手 Spring Security

  • pom.xml

    <parent>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-parent</artifactId>
    <version>2.1.5.RELEASE</version>
</parent>
<dependencies>
    <!--mvc依赖-->
    <dependency>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-web</artifactId>
    </dependency>
    <!--用到前端，加个前端模板-->
    <dependency>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-thymeleaf</artifactId>
    </dependency>
    <dependency>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-security</artifactId>
    </dependency>
</dependencies>

权限管理

定义两个 HTML 资源：index.html、admin.html，

同时定义两个⻆⾊ ADMIN 和 USER

  1. ADMIN 拥有 访问 index.html 和 admin.html 的权限

  2. USER 只有访问 index.html 的权限。

  • 创建 SecurityConfig 类，管理类。

    package com.southwind.config;

import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.authentication.builders.AuthenticationManagerBuilder;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;

/*
    通过继承方式，然后重写权限分配的方法
    控制角色和对应权限
 */
@Configuration//表示他是一个配置的类
@EnableWebSecurity//让WebSecurity生效
public class SecurityConfig extends WebSecurityConfigurerAdapter {
    @Override
    //这是设置账户和角色的关系
    protected void configure(AuthenticationManagerBuilder auth) throws Exception {
        //链式编程
        auth.inMemoryAuthentication()
                //添加密码编译器
                .passwordEncoder(new MyPasswordEncoder())
                //添加用户名
                .withUser("user")
                //添加一个编译过后的密码
                .password(new MyPasswordEncoder().encode("000"))
                //分配User角色
                .roles("USER")
                //继续添加其他用户
                .and()
                .withUser("admin")
                //添加一个编译过后的密码
                .password(new MyPasswordEncoder().encode("111"))
                //分配USER和ADMIN角色
                .roles("USER","ADMIN");
    }

    //设置角色和权限的关系
    @Override
    protected void configure(HttpSecurity http) throws Exception {
        //对请求进行一个拦截，设置hasRole权限
        http.authorizeRequests()
                .antMatchers("/admin").hasRole("ADMIN")//单个角色用hasRole，多个用access
                .antMatchers("/index").access("hasRole('ADMIN') or hasRole('USER')")
                .anyRequest().authenticated()//以上拦截请求均需要权限认证
            .and()
                //设置自定义login页面
                .formLogin()
                .loginPage("/login")
                .permitAll()//登录页面无需认证
            .and()
                //设置自定义logout
                .logout()
                .permitAll()//注销页面无需认证
            .and()
                .csrf()
                .disable();//禁用csrf模块
    }
}

  • 创建 MyPasswordEncoder 密码的编译类

    package com.southwind.config;

import org.springframework.security.crypto.password.PasswordEncoder;

/*
    通过实现接口方式，然后重写对密码的编码

 */
public class MyPasswordEncoder implements PasswordEncoder {
    @Override
    public String encode(CharSequence charSequence) {
        return charSequence.toString();
    }

    @Override
    public boolean matches(CharSequence charSequence, String s) {
        //String s 是密码，检测密码和charSequence参数是否匹配
        return s.equals(charSequence.toString());
    }
}

  • 修改Handler

    package com.southwind.controller;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.GetMapping;
@Controller
public class HelloHandler {

    //USER和ADMIN能访问
    @GetMapping("/index")
    public String index(){
        return "index";
    }

    //只有ADMIN能访问的页面
    @GetMapping("/admin")
    public String admin(){
        return "admin";
    }

    //自定义的login page
    @GetMapping("/login")
    public String login(){
        return "login";
    }
}

  • login.html

    <!DOCTYPE html>
<html xmlns:th="http://www.thymeleaf.org">
<html lang="en">
  <head>
    <meta charset="UTF-8">
    <title>Title</title>
  </head>
  <body>
    <!-- 项目名不确定，动态跳转对应security验证功能 -->
    <form th:action="@{/login}" method="post">
      <!-- 交由Spring Security处理识别，所以username和password名字不能更改！！-->
      ⽤户名：<input type="text" name="username"/><br/>
      密码：<input type="text" name="password"/><br/>
      <input type="submit" value="登录"/>
    </form>
  </body>
</html>
